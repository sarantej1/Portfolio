-- Banking Transaction Intelligence System | MySQL 8.0 CREATE TABLE bank_transactions ( transaction_id VARCHAR(10) PRIMARY KEY, customer_id VARCHAR(10), age INT, city VARCHAR(50), amount DECIMAL(15,2), status VARCHAR(20) ); -- Month-over-Month Growth (Window Function) WITH monthly_volume AS ( SELECT DATE_FORMAT(transaction_date,'%Y-%m') month, ROUND(SUM(amount),2) total_volume FROM bank_transactions WHERE status='Success' GROUP BY month ) SELECT month, total_volume, LAG(total_volume) OVER(ORDER BY month) prev_month, ROUND((total_volume - LAG(total_volume)OVER(ORDER BY month)) * 100 / NULLIF(LAG(total_volume)OVER(ORDER BY month),0), 2) AS mom_growth_pct FROM monthly_volume; -- Jan:₹18.5L Feb:₹19.8L Mar:₹30.5L Apr:₹24.2L -- Customer Lifetime Value (CTE) WITH customer_stats AS ( SELECT customer_id, customer_name, COUNT(*) num_transactions, ROUND(SUM(amount),2) total_volume, DATEDIFF(MAX(transaction_date),MIN(transaction_date)) lifespan_days FROM bank_transactions WHERE status='Success' GROUP BY customer_id, customer_name ) SELECT *, ROUND(total_volume/NULLIF(lifespan_days,0)*365,2) annualised_value FROM customer_stats ORDER BY total_volume DESC;

from statsmodels.tsa.arima.model import ARIMA from statsmodels.graphics.tsaplots import plot_acf, plot_pacf df = pd.read_csv("ev_charging_data.csv", parse_dates=['date'], index_col='date') ts = df['charging_demand'] # ACF/PACF to determine optimal (p, d, q) fig, axes = plt.subplots(1, 2, figsize=(14,4)) plot_acf(ts.diff().dropna(), ax=axes[0], lags=30) plot_pacf(ts.diff().dropna(), ax=axes[1], lags=30) # Fit ARIMA Model model = ARIMA(ts, order=(2, 1, 2)) result = model.fit() print(result.summary()) # 12-period Forecast with 95% Confidence Intervals forecast = result.get_forecast(steps=12) forecast_df = forecast.summary_frame() plt.plot(ts, label='Historical Demand') plt.plot(forecast_df['mean'], label='Forecast', color='red') plt.fill_between(forecast_df.index, forecast_df['mean_ci_lower'], forecast_df['mean_ci_upper'], alpha=0.2, color='red', label='95% CI') plt.title('EV Charging Demand Forecast') plt.legend(); plt.tight_layout()
